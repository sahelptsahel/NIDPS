⚠️ Prerequisites
bash# Run these on Kali, targeting the machine running NIDPS
TARGET="192.168.1.x"   # replace with your NIDPS machine IP

# Tools needed (all pre-installed on Kali)
sudo apt install nmap hping3 scapy ettercap-text-only -y

1. 🔍 Port Scan
bash# Fast SYN scan across all ports — triggers 20+ unique ports threshold
sudo nmap -sS -T4 -p- $TARGET

# Aggressive scan with version detection
sudo nmap -A -T5 $TARGET

# Quick top-1000 ports
nmap -T4 $TARGET

2. 💥 SYN Flood
bash# hping3 SYN flood — 100+ SYN packets in 5 seconds
sudo hping3 -S --flood -V -p 80 $TARGET

# With random source IPs (spoofed)
sudo hping3 -S --flood --rand-source -p 80 $TARGET

# Target specific port with rate control
sudo hping3 -S -p 443 --faster $TARGET

3. 🌊 UDP Flood (DDoS)
bash# UDP flood on port 53
sudo hping3 --udp --flood -p 53 $TARGET

# Random destination ports
sudo hping3 --udp --flood --rand-dest $TARGET

4. 🏓 ICMP Flood (DDoS)
bash# Ping flood
sudo hping3 --icmp --flood $TARGET

# Standard ping flood
sudo ping -f $TARGET

# With large packet size
sudo hping3 --icmp --flood -d 1400 $TARGET

5. 🎄 XMAS Scan
bash# Sets FIN + PSH + URG flags simultaneously
sudo nmap -sX -T4 $TARGET

# With hping3
sudo hping3 -F -P -U -p 80 $TARGET

# Full port range
sudo nmap -sX -p- $TARGET

6. 🕳️ NULL Scan
bash# No flags set on TCP packets
sudo nmap -sN -T4 $TARGET

# With hping3 (no flags)
sudo hping3 -p 80 $TARGET

# Full port range
sudo nmap -sN -p- $TARGET
